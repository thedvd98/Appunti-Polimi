\subsubsection{Per sostituzione}
\begin{equation*}
	\int f(g(x)) g'(x)\ud x
\end{equation*}
$u=g(x) \text{ e }du = g(x)'dx$
\begin{equation*}
	\int f(u) \ud u
\end{equation*}
\subsubsection{Per parti}
\begin{equation}
	\int f'(x) g(x) \mathrm{d}x = f(x)g(x) - \int f(x) g'(x)\ud x
\end{equation}
Per risolvere tipo $\int \ln(x) \ud x$ o $\int x \sin(x) \ud x$
